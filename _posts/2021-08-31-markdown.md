#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

//常用的查找算法 find

//查找内置数据类型 例如vector

void test01()
{
    vector<int> v;
    for (int i = 0; i < 10; i++)
    {
        v.push_back(i);
    }
    //查找容器中是否有5这个元素
    vector<int>::iterator it = find(v.begin(), v.end(), 50);
    if (it == v.end())
    {
        cout << "没有找到！" << endl;
    }
    else
    {
        cout << "找到了！" << endl;
    }
}
